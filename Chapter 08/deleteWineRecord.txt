    func deleteWineRecord(record:String){        let deleteSQL = "DELETE FROM wine WHERE name = ?"                if(sqlite3_open(dbPath.path!, &db)==SQLITE_OK){            if(sqlite3_prepare_v2(db, deleteSQL, -1, &sqlStatement, nil)==SQLITE_OK){                sqlite3_bind_text(sqlStatement, 1, record, -1, SQLITE_TRANSIENT)                if(sqlite3_step(sqlStatement)==SQLITE_DONE){                    print("item deleted")                }else{                    print("unable to delete")                }            }                    }    }
