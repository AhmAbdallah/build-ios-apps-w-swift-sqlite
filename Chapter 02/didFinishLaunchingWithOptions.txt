func application(application: UIApplication, didFinishLaunchingWithOptions launchOptions: [_ NSObject: AnyObject]?) -> Bool {//….Code remove for brevity   ….        var srcPath:URL        var destPath:URL        let dirManager = FileManager.default()        let projectBundle = Bundle.main()        do {            let resourcePath = projectBundle.pathForResource("Chapter2", ofType: "sqlite")            let documentURL = try dirManager.urlForDirectory(FileManager.SearchPathDirectory.documentDirectory, in: FileManager.SearchPathDomainMask.userDomainMask, appropriateFor: nil, create: true)                        srcPath = URL(fileURLWithPath: resourcePath!)            destPath = documentURL.appendingPathComponent("Chapter2.sqlite")                        if !dirManager.fileExists(atPath:destPath.path!) {               try dirManager.copyItem(at: srcPath, to: destPath)            }                    } catch let err as NSError {            print("Error: \(err.domain)")        }            }
